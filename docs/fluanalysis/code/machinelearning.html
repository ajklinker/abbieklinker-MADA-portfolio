Null Model

Build the model

null<-null_model() %>% 
  set_engine("parsnip") %>% 
  set_mode("regression") %>% 
  translate()

null_wf <-
  workflow() %>%
  add_model(null) %>%
  add_recipe(model_recipe3)

null_fit <-
  null_wf %>%
  fit(train_data)
